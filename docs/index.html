Diff Type

Rich datetime differences with calendar-aware and precise calculations

// Create difference between dates
start := chronogo.Date(2023, time.January, 15)
end := chronogo.Date(2025, time.March, 20)
diff := start.Diff(end)

// Calendar-aware differences
fmt.Printf("%d years, %d months\n", diff.Years(), diff.Months())

// Precise differences
fmt.Printf("%.2f years\n", diff.InYears())
fmt.Printf("%d total days\n", diff.InDays())

// Human-readable
fmt.Println(diff.ForHumans())  // "2 years from now"


Calendar-aware: 2 years, 2 months
Precise: 2.18 years, 795 days
Human: "2 years from now"
Comprehensive comparison methods

Timezone Operations

Seamless timezone conversions and management

// Create and convert timezones

fmt.Println("Tokyo:", tokyo.Format("15:04"))
fmt.Println("London:", london.Format("15:04"))

Period Operations

Work with time ranges and intervals

// Create a period between two dates
start := chronogo.Date(2025, time.January, 1)
end := chronogo.Date(2025, time.January, 31)
period := chronogo.NewPeriod(start, end)

// Iterate by day
for _, dt := range period.RangeByUnitSlice(chronogo.Day, 1) {
    fmt.Println(dt.Format("Jan 2"))
}

// Check if date is in period
inRange := period.Contains(chronogo.Date(2025, time.January, 15))


Jan 1, Jan 2, Jan 3, ... Jan 31
Contains check: true
Flexible iteration intervals

Advanced Parsing

ISO 8601 extensions and custom format tokens

// ISO 8601 Ordinal dates (day of year)
ordinal, _ := chronogo.Parse("2023-359")     // Dec 25, 2023

// ISO 8601 Week dates  
week, _ := chronogo.Parse("2023-W52-1")      // Monday of week 52

// Token-based custom formats
date, _ := chronogo.FromFormatTokens(
    "25/12/2023 14:30", 
    "DD/MM/YYYY HH:mm"
)

// ISO 8601 Intervals
interval, _ := chronogo.Parse(
    "2023-01-01T00:00:00Z/2023-12-31T23:59:59Z"
)


Ordinal: 359th day = Dec 25
Week: W52-1 = Monday of week 52
Tokens: Custom format parsing
Intervals: Date range support

Comparison Methods

Intuitive date comparisons and validations

// Date comparisons
dt := chronogo.Parse("2025-12-25")

// Time checks
isFuture := dt.IsFuture()
isPast := dt.IsPast()
isToday := dt.IsToday()
isWeekend := dt.IsWeekend()

// Range checks
now := chronogo.Now()
inRange := dt.Between(now, now.AddDays(30), true)

// Relative comparisons

farthest := chronogo.Farthest(now, dt1, dt2, dt3)

Holiday-Aware Scheduler

Intelligent scheduling that respects holidays

// Create scheduler
scheduler := chronogo.NewHolidayAwareScheduler("US")

// Schedule recurring meetings (avoids holidays)
start := chronogo.Date(2025, time.September, 1)
meetings := scheduler.ScheduleRecurring(
    start, 
    7*24*time.Hour, 
    8, // 8 occurrences
)

// Monthly end-of-month reports
reports := scheduler.ScheduleMonthlyEndOfMonth(start, 6)

// Business days only
bizMeetings := scheduler.ScheduleBusinessDays(start, 10)
